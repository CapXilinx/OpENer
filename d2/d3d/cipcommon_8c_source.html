00073 
00074   /* find the instance: if instNr==0, the class is addressed, else find the instance */
00075   EipUint16 instance_number = message_router_request->request_path.instance_number; /* get the instance number */

00081

00083     if (NULL != service) /* if services are defined */

00085       for (int i = 0; i < instance->cip_class->number_of_services; i++) /* seach the services list */
00086       {

00088         {

00091           OPENER_ASSERT(NULL != service->service_function);

00102   } else {

00104     /* if instance not found, return an error reply*/

00117 CipInstance *AddCipInstances(CipClass *RESTRICT const cip_class, const int number_of_instances) {

00119   EipUint32 instance_number = 1; /* the first instance is number 1 */
00120 
00121   OPENER_TRACE_INFO("adding %d instances to class %s\n", number_of_instances,
00122                     cip_class->class_name);
00123 
00124   next_instance = &cip_class->instances; /* get address of pointer to head of chain */
00125   while (*next_instance) /* as long as what pp points to is not zero */
00126   {
00127     next_instance = &(*next_instance)->next; /*    follow the chain until pp points to pointer that contains a zero */
00128     instance_number++; /*    keep track of what the first new instance number will be */
00129   }
00130 
00131   CipInstance *current_instance = current_instance = (CipInstance *) CipCalloc(
00132         number_of_instances, sizeof(CipInstance)); /* allocate a block of memory for all created instances*/
00133   CipInstance *first_instance = current_instance; /* allocate a block of memory for all created instances*/
00134 
00135   OPENER_ASSERT(NULL != current_instance);
00136   /* fail if run out of memory */
00137 
00138   cip_class->number_of_instances += number_of_instances; /* add the number of instances just created to the total recorded by the class */
00139 
00140   for (int i = 0; i < number_of_instances; i++) /* initialize all the new instances */
00141   {
00142     *next_instance = current_instance; /* link the previous pointer to this new node */
00143 
00144     current_instance->instance_number = instance_number; /* assign the next sequential instance number */
00145     current_instance->cip_class = cip_class; /* point each instance to its class */
00146

00151     }
00152

00154     instance_number++; /* update to the number of the next node*/

00287

00291   for (int i = 0; i < instance->cip_class->number_of_attributes; i++) {
00292     if (attribute->data == NULL) { /* found non set attribute */

00294       attribute->type = cip_type;
00295       attribute->attribute_flags = cip_flags;

00297 
00298       if (attribute_number > instance->cip_class->highest_attribute_number) /* remember the max attribute number that was defined*/

00305   }
